## Lab Build Systems Example  
Makefile
```
all: static_block dynamic_block

block.o: source/block.c
		cc -c source/block.c -o block.o
block.so: block.o
		cc -shared -o block.so block.o
block.a: block.o
		ar qc block.a block.o

clean:
		rm dynamic_block static_block program.o block.o block.a block.so

dynamic_block: program.o block.so
		cc program.o block.so -o dynamic_block -Wl,-rpath,'$$ORIGIN'

static_block: program.o block.a
		cc program.o block.a -o static_block -Wl,-rpath,'$$ORIGIN'

```  

CMakeLists.txt  
```
cmake_minimum_required(VERSION 3.10)

# set the project name
project(static_block VERSION 1.0)
project(dynamic_block VERSION 1.0)

add_subdirectory(source)

# add the executable
add_executable(static_block program.c)
add_executable(dynamic_block program.c)

target_link_libraries(static_block PUBLIC StaticBlock)
target_link_libraries(dynamic_block PUBLIC DynamicBlock)

# add the binary tree to the search path for include files
# so that we will find TutorialConfig.h
target_include_directories(static_block PUBLIC
                          "${PROJECT_BINARY_DIR}"
                          "${PROJECT_SOURCE_DIR}/source"
                          "${PROJECT_SOURCE_DIR}/headers"
                          )

target_include_directories(dynamic_block PRIVATE
                          "${PROJECT_BINARY_DIR}"
                          "${PROJECT_SOURCE_DIR}/source"
                          "${PROJECT_SOURCE_DIR}/headers"
                          )
```
